create table if not exists contest_config (
  key text primary key,
  value text
);

create table if not exists entries (
  id uuid default gen_random_uuid() primary key,
  discord_name text not null,
  server_name text not null,
  picks jsonb not null,
  score integer default 0,
  submitted_at timestamptz default now()
);

alter table entries enable row level security;
alter table contest_config enable row level security;

drop policy if exists "public_insert" on entries;
drop policy if exists "public_select" on entries;
drop policy if exists "public_update" on entries;
drop policy if exists "public_all" on contest_config;

create policy "public_insert" on entries for insert with check (true);
create policy "public_select" on entries for select using (true);
create policy "public_update" on entries for update using (true);
create policy "public_all" on contest_config for all using (true) with check (true);
